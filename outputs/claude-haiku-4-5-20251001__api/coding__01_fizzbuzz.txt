def solve(n):
    result = []
    for i in range(1, n + 1):
        output = ""
        if i % 3 == 0:
            output += "fizz"
        if i % 5 == 0:
            output += "buzz"
        if i % 7 == 0:
            output += "bang"
        if output == "":
            output = str(i)
        result.append(output)
    return result
