def solve(n):
    result = []
    for i in range(1, n + 1):
        s = ''
        if i % 3 == 0:
            s += 'fizz'
        if i % 5 == 0:
            s += 'buzz'
        if i % 7 == 0:
            s += 'bang'
        if not s:
            s = str(i)
        result.append(s)
    return result
