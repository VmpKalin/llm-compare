```sql
SELECT 
    c.country,
    COUNT(DISTINCT ac.customer_id) AS customer_count,
    SUM(o.total_cents) / 100.0 AS total_revenue_euros
FROM (
    SELECT DISTINCT customer_id
    FROM orders
    WHERE created_at >= DATE '2025-01-01' AND created_at < DATE '2026-01-01'
) AS ac
JOIN customers c ON ac.customer_id = c.id
JOIN orders o ON ac.customer_id = o.customer_id
GROUP BY c.country
ORDER BY total_revenue_euros DESC;
```
