```sql
SELECT 
    c.country,
    COUNT(DISTINCT o.customer_id) AS customer_count,
    SUM(o.total_cents) / 100.0 AS total_revenue_euros
FROM 
    orders o
JOIN 
    customers c ON o.customer_id = c.id
WHERE 
    EXTRACT(YEAR FROM o.created_at) = 2025
GROUP BY 
    c.country
ORDER BY 
    total_revenue_euros DESC;
```
